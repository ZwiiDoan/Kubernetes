apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: db-policy
spec:
  podSelector:
    matchLabels:
      role: db
  policyTypes:
  - Ingress
  ingress:
  - from:
    - podSelector:
        matchLabels: name: api-prod
    ports:
    - protocol: TCP
      port: 3306

---
kubectl create -f policy-definition.yaml

---
# Flannel does not support Network Policies
